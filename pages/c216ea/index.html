谷粒商城下载的sql中,pmsattr,缺少valuetype字段

自己加上。。

ALTER TABLE `elitemall_pms`.`pms_attr` 
ADD COLUMN `value_type` int NULL COMMENT '值类型[0 - 单选，1 - 多选]' AFTER `icon`;
